        SQL Reference

        SELECT Statement

        SELECT column1, column2
FROM table_name
WHERE condition
ORDER BY column
LIMIT count;

        JOIN Types

        INNER JOIN - matching rows only
LEFT JOIN - all left + matching
RIGHT JOIN - all right + matching
FULL OUTER JOIN - all rows

        Aggregate Functions

        COUNT() - count rows
SUM() - sum values
AVG() - average value
MIN() - minimum value
MAX() - maximum value

        GROUP BY

        SELECT column, COUNT(*)
FROM table
GROUP BY column
HAVING COUNT(*) > 1;

        Subquery

        SELECT * FROM employees
WHERE salary > (
    SELECT AVG(salary) FROM employees
);

        Window Function

        SELECT name, salary,
       RANK() OVER (ORDER BY salary DESC)
FROM employees;
